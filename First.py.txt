import matplotlib.pyplot as plt
import numpy as np
from sklearn import datasets,linear_model
house_price_inMillions=[12,3,9,7,20]
house_size=[1,2,3,4,5]
house_size2=np.array(house_size).reshape((-1,1))
#print(house_size2)
regr=linear_model.LinearRegression()
regr.fit(house_size2,house_price_inMillions)
print("Coef : ",regr.coef_)
print("Intercept : ",regr.intercept_)
size1=11
price1=size1*regr.coef_+regr.intercept_
#print(price1)
def graph(formula,x_rng):
    x=np.array(x_rng)
    y=eval(formula)
    plt.plot(x,y)
    
graph('x*regr.coef_+regr.intercept_',range(1,10))
plt.scatter(house_size,house_price_inMillions,color='orange')
plt.ylabel('Price')
plt.xlabel('Size')
plt.show()
